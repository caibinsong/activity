examples

基础功能展示

<div class="my-rating"></div>

$(".my-rating").starRating();

圆角星星[且绑定date-rating到元素上来初始化星星个数]

<div> class="my-rating-2" data-rating="2.5"></div>

$(".my-rating-2").starRating({
	totalStars: 5,
	starShape: 'rounded',
	starSize: 40,
	emptyColor: 'lightgray',
	hoverColor: 'salmon',
	activeColor: 'crimson',
	useGradient: false
	});

使用渐变色

<div> class="my-rating-3"></div>

// specify the gradient start and end for the selected stars
$(".my-rating-3").starRating({
	starSize: 80,
	strokeWidth: 9,
	strokeColor: 'black',
	initialRating: 2,
	starGradient: {
		start: '#93BFE2',
		end: '#105694'
	},
	useGradient: true
});

评级选定之后的自定义回调函数

<div> class="my-rating-4"></div>

$(".my-rating-4").starRating({
	totalStars: 5,
	emptyColor: 'lightgray',
	hoverColor: 'salmon',
	activeColor: 'cornflowerblue',
	initialRating: 4,
	strokeWidth: 0,
	useGradient: false,
	callback: function(currentRating, $el){
		alert('rated ', currentRating);
		console.log('DOM element ', $el);
	}
});

只读模式[不能交互，只能看]

<div> class="my-rating-5"></div>

$(".my-rating-5").starRating({
	readOnly: true
});

使用整星评定(不使用半星评定)

// use full stars only
<div> class="my-rating-6"></div>

$(".my-rating-6").starRating({
	useFullStars: true
});

hover事件回调

<span> class="my-rating-7"></span>
<span> class="live-rating"></span>

$(".my-rating-7").starRating({
	initialRating: 3.5,
	disableAfterRate: false,
	onHover: function(currentIndex, currentRating, $el){
		$('.live-rating').text(currentIndex);
	},
	onLeave: function(currentIndex, currentRating, $el){
		$('.live-rating').text(currentRating);
	}
});
